OOCSS

/* Without OOCSS Methodology */

  width: 100px;
  border: 1px solid black;
}

.widget {
  width: 200px;
  overflow: hidden;
  border: 1px solid black;
}
                
              
                  
/* After applying OOCSS strategy */

.skin {
  border: 1px solid black;
}

.button {
  width: 100px;
}

.widget {
  width: 200px;
  overflow: hidden;
}
